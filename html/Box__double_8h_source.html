190  /**

195  bool empty() const {
196  //assert(mLimits.first.dimension() == mLimits.second.dimension());
197  if ( mLimits.first.dimension() == 0 ) {
198  return true;
199  }
200  for ( std::size_t d = 0; d < mLimits.first.dimension(); ++d ) {
201  if ( mLimits.first.at(d) > mLimits.second.at(d) ) {
202  return true;
203  }
204  }
205  return false;
206  }

208  /**
209  * @brief Determines if the current box is symmetric.
210  * @details A box is symmetric, if its centroid is the origin in the respective space, i.e. if the center of each interval in each
211  * dimension is zero.
212  * @return True if symmetric, False otherwise.
213  */
214  bool isSymmetric() const {
215  if ( mLimits.first.dimension() == 0 ) {
216  return true;
217  }
218  for ( std::size_t d = 0; d < mLimits.first.dimension(); ++d ) {
219  if ( mLimits.first.at(d) != -mLimits.second.at(d) ) {
220  return false;
221  }
222  }
223  return true;
224  }
