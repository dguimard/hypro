117  void addConstraint(const vector_t<Number>& normal, Number offset) {
118  if(mConstraints.cols() == 0) {
119  mConstraints = matrix_t<Number>(normal);
120  mConstants = vector_t<Number>::Zero(1);
121  mConstants << offset;
122  } else {
123  assert(mConstraints.cols() == normal.rows());
124  mConstraints.conservativeResize(mConstraints.rows()+1, mConstraints.cols());
125  mConstants.conservativeResize(mConstraints.rows()+1);
126 
127  mConstraints.row(mConstraints.rows()-1) = normal;
128  mConstants(mConstants.rows()-1) = offset;
129  }
130  mIsBox = TRIBOOL::NSET;
