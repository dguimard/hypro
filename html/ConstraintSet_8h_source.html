114  void addConstraint(const vector_t<Number>& normal, Number offset) {
115  if(mConstraints.cols() == 0) {
116  mConstraints = matrix_t<Number>(normal);
117  mConstants = vector_t<Number>::Zero(1);
118  mConstants << offset;
119  } else {
120  assert(mConstraints.cols() == normal.rows());
121  mConstraints.conservativeResize(mConstraints.rows()+1, mConstraints.cols());
122  mConstants.conservativeResize(mConstraints.rows()+1);
123 
124  mConstraints.row(mConstraints.rows()-1) = normal;
125  mConstants(mConstants.rows()-1) = offset;
126  }
127  mIsBox = TRIBOOL::NSET;
