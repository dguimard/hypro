102  if(mConstraints.cols() == 0) {
103  mConstraints = matrix_t<Number>(normal);
104  mConstants = vector_t<Number>::Zero(1);
105  mConstants << offset;
106  } else {
107  assert(mConstraints.cols() == normal.rows());
