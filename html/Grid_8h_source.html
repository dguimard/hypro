25 struct pointIt {
26  typename std::vector<std::vector<Number>> v;
27  std::vector<typename std::vector<Number>::const_iterator> it;
28  bool end = false;
29 
30  pointIt( const std::vector<std::vector<Number>>& _v )
31  : v( _v ) {
32  for ( std::size_t i = 0; i < v.size(); ++i ) it.push_back( v.at( i ).begin() );
33 
34  end = false;
35  }
36 
37  // prefix
38  Point<Number> operator++() {
39  if ( end ) return std::move( current() );
40  ++it.back();
41  std::size_t i = it.size() - 1;
42  while ( it.at( i ) == v.at( i ).end() ) {
43  it[i] = v.at( i ).begin();
44  if ( i == 0 ) {
45  end = true;
46  return std::move( current() );
47  }
48  --i;
49  it[i]++;
50  }
51  return current();
52  }
53 
54  // postfix
55  Point<Number> operator++( int ) { return ++( *this ); }
56 
57  Point<Number> current() {
58  assert( v.size() == it.size() );
59  Point<Number> res( vector_t<Number>::Zero( v.size() ) );
60  for ( std::size_t i = 0; i < v.size(); ++i ) res[i] = *it.at( i );
61 
62  return res;
63  }
