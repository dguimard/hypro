323  template <typename N = Number, carl::DisableIf<std::is_same<N, double>> = carl::dummy>
324  void reduceNumberRepresentation( const std::vector<Point<Number>>& _vertices = std::vector<Point<Number>>(), unsigned limit = fReach_DENOMINATOR ) const {
325  //#ifdef REDUCE_NUMBERS
326  if ( Setting::REDUCE_NUMBERS == true ) {
327  TRACE( "hypro.hPolytope", "Attempt to reduce numbers." );
328  std::vector<Point<Number>> originalVertices;
329  if ( _vertices.empty() ) {
330  TRACE( "hypro.hPolytope", "No passed vertices, computed vertices." );
331  originalVertices = this->vertices();
332  } else {
333  TRACE( "hypro.hPolytope", "Use passed vertices." );
334  originalVertices = _vertices;
335  }
336  TRACE( "hypro.hPolytope", "Vertices empty: " << originalVertices.empty() );
337 
338  if ( !this->empty() ) {
339  // normal reduction
340  for ( unsigned planeIndex = 0; planeIndex < mHPlanes.size(); ++planeIndex ) {
341 #ifdef HPOLY_DEBUG_MSG
342  std::cout << "Original: " << mHPlanes.at( planeIndex ) << std::endl;
343 #endif
344  // find maximal value
345  Number largest = Number( 0 );
346  mHPlanes.at( planeIndex ).makeInteger();
347 #ifdef HPOLY_DEBUG_MSG
348  std::cout << "As Integer: " << mHPlanes.at( planeIndex ) << std::endl;
349 #endif
350  largest = carl::abs( mHPlanes.at( planeIndex ).offset() );
351  for ( unsigned i = 0; i < mDimension; ++i ) {
352  if ( carl::abs( mHPlanes.at( planeIndex ).normal()( i ) ) > largest ) {
353  largest = carl::abs( mHPlanes.at( planeIndex ).normal()( i ) );
354  }
355  }
356 
357  // reduce, if reduction is required
358  if ( largest > ( limit * limit ) ) {
359 #ifdef HPOLY_DEBUG_MSG
360  std::cout << "Actual reduction" << std::endl;
361 #endif
362  vector_t<Number> newNormal( mDimension );
363  for ( unsigned i = 0; i < mDimension; ++i ) {
364  newNormal( i ) = carl::floor( Number( ( mHPlanes.at( planeIndex ).normal()( i ) / largest ) * Number( limit ) ) );
365  assert( carl::abs( Number( mHPlanes.at( planeIndex ).normal()( i ) / largest ) ) <= Number( 1 ) );
366  assert( carl::isInteger( newNormal( i ) ) );
367  assert( newNormal( i ) <= Number( limit ) );
368  }
369  mHPlanes.at( planeIndex ).setNormal( newNormal );
370  Number newOffset = mHPlanes.at( planeIndex ).offset();
371  newOffset = carl::ceil( Number( ( newOffset / largest ) * Number( limit ) ) );
372  for ( const auto& vertex : originalVertices ) {
373  Number tmp = newNormal.dot( vertex.rawCoordinates() );
374  if ( tmp > newOffset ) {
375  newOffset = newOffset + ( tmp - newOffset );
376  assert( newNormal.dot( vertex.rawCoordinates() ) <= newOffset );
377  }
378  assert( Halfspace<Number>( newNormal, newOffset ).contains( vertex ) );
379  }
380  newOffset = carl::ceil( newOffset );
381 #ifdef HPOLY_DEBUG_MSG
382  std::cout << "Reduced to " << convert<Number, double>( newNormal ).transpose() << " <= " << carl::toDouble( newOffset ) << std::endl;
383 #endif
384  mHPlanes.at( planeIndex ).setOffset( newOffset );
385  }
386 #ifdef HPOLY_DEBUG_MSG
387  std::cout << "Reduced: " << mHPlanes.at( planeIndex ) << std::endl;
388 #endif
389  }
390 #ifdef HPOLY_DEBUG_MSG
391 //std::cout << "After Reduction: " << *this << std::endl;
392 #endif
393  }
394  }
395  //#endif
396  }
