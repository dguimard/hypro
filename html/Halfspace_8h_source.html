96  template<typename N = Number, carl::DisableIf< std::is_same<N,double> > = carl::dummy>
97  void makeInteger() {
98  if(!mIsInteger){
99  Number scaling = Number(carl::getDenom(mScalar));
100  for(unsigned i = 0; i < mNormal.rows(); ++i) {
101  scaling = scaling * Number(carl::getDenom(mNormal(i)));
102  }
103 
104  mScalar = mScalar*scaling;
105  assert(carl::isInteger(mScalar));
106 
107  for(unsigned i = 0; i < mNormal.rows(); ++i) {
108  mNormal(i) = mNormal(i)*scaling;
109  assert(carl::isInteger(mNormal(i)));
110  }
111  mIsInteger = true;
112  }
113  }

377  template<class Number>
378  struct hash<hypro::Halfspace<Number>> {
379  std::size_t operator()(hypro::Halfspace<Number> const& Halfspace) const
380  {
381  size_t seed = 0;
382  hypro::vector_t<Number> normal = Halfspace.normal();
383  Number scalar = Halfspace.scalar();
384  std::hash<hypro::vector_t<Number>> vectorHasher;
