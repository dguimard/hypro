667 int effectiveDimension(const std::vector<Point<Number>>& vertices) {
668  if(!vertices.empty()){
669  if(vertices.size() == 1) {
670  return 0;
671  }
672  long maxDim = vertices.begin()->rawCoordinates().rows();
673  matrix_t<Number> matr = matrix_t<Number>(vertices.size()-1, maxDim);
674  // use first vertex as origin, start at second vertex
675  long rowIndex = 0;
676  for(auto vertexIt = ++vertices.begin(); vertexIt != vertices.end(); ++vertexIt, ++rowIndex) {
677  matr.row(rowIndex) = (vertexIt->rawCoordinates() - vertices.begin()->rawCoordinates()).transpose();
678  }
679  return int(matr.fullPivLu().rank());
680  }
681  return -1;
682 }
