655  if(!vertices.empty()){

657  return 0;

660  matrix_t<Number> matr = matrix_t<Number>(vertices.size()-1, maxDim);

664  matr.row(rowIndex) = (vertexIt->rawCoordinates() - vertices.begin()->rawCoordinates()).transpose();
665  }
