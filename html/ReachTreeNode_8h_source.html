82  initialSet.setSet(boost::get<VPolytope<Number>>(set));
83  break;
84  }
85  #ifdef HYPRO_USE_PPL
86  case representation_name::ppl_polytope: {
87  initialSet.setSet(boost::get<Polytope<Number>>(set));
88  break;
89  }
90  #endif
91  case representation_name::support_function: {
92  initialSet.setSet(boost::get<SupportFunction<Number>>(set));
93  break;
94  }
95  case representation_name::taylor_model: {
96  assert(false);

100  initialSet.setSet(boost::get<Zonotope<Number>>(set));
