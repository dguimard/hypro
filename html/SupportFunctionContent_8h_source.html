48 struct trafoContent {
49  std::shared_ptr<SupportFunctionContent<Number, Setting>> origin;
50  std::shared_ptr<const lintrafoParameters<Number>> parameters;
51  unsigned currentExponent;
52  std::size_t successiveTransformations;
53  // 2^power defines the max. number of successive lin.trans before reducing the SF
54 
55  trafoContent( const std::shared_ptr<SupportFunctionContent<Number, Setting>>& _origin, const matrix_t<Number>& A, const vector_t<Number>& b )
56  : origin( _origin )
57  , currentExponent( 1 ) {
58  // Determine, if we need to create new parameters or if this matrix and vector pair has already been used (recursive).
59  parameters = std::make_shared<const lintrafoParameters<Number>>( A, b );
60  // in case this transformation has already been performed, parameters will be updated.
61  auto trafoInDepth = _origin->hasTrafo( parameters, A, b, 0 ); // returns a pair of bool and depth in which the node was found -> this saves some comparisons later, as we know if the second value is 1, we can reduce.
62  assert( trafoInDepth.second != 1 || ( ( origin->type() == SF_TYPE::LINTRAFO ) && ( *origin->linearTrafoParameters()->parameters == *parameters ) && trafoInDepth.first ) );
63  if ( Setting::USE_LIN_TRANS_REDUCTION ) {
64  DEBUG( "hypro.representations.supportFunction", "Reduction of sf with depth: " << origin->depth() + 1 );
65  // best points for reduction are powers of 2 thus we only use these points for possible reduction points
66  bool reduced;
67  do {
68  reduced = false;
69  if ( trafoInDepth.first && trafoInDepth.second == 1 && origin->linearTrafoParameters()->currentExponent == currentExponent ) {
70  successiveTransformations = origin->linearTrafoParameters()->successiveTransformations + 1;
71  } else {
72  successiveTransformations = 0;
73  }
74  TRACE( "hypro.representations.supportFunction", "successiveTransformations with exponent " << currentExponent << ": " << successiveTransformations );
75  if ( successiveTransformations == unsigned( carl::pow( 2, parameters->power ) - 1 ) ) {
76  TRACE( "hypro.representations.supportFunction", "Update origin pointer as a result of lintrafo reduction." );
77  reduced = true;
78  currentExponent = currentExponent * ( carl::pow( 2, parameters->power ) );
79  for ( std::size_t i = 0; i < unsigned( carl::pow( 2, parameters->power ) - 1 ); i++ ) {
80  origin = origin->linearTrafoParameters()->origin;
81  }
82  // Note: The following assertion does not hold in combination with the current reduction techniques.
83  //assert(origin->type() != SF_TYPE::LINTRAFO || (origin->linearTrafoParameters()->parameters == this->parameters && origin->linearTrafoParameters()->currentExponent >= currentExponent) );
84 
85  // update information whether another trafo follows/is the origin.
86  // TODO: If the next operation is not a linTrafo we can directly quit.
87  trafoInDepth = origin->hasTrafo( parameters, A, b, 0 );
88  TRACE( "hypro.representations.supportFunction", "Next linear trafo at depth: " << trafoInDepth.second );
89  TRACE( "hypro.representations.supportFunction", "Next node type: " << origin->type() );
90  }
91  } while ( reduced == true );
92  DEBUG( "hypro.representations.supportFunction", "Reduced sf to depth: " << origin->depth() + 1 );
93  assert( origin->checkTreeValidity() );
94  }
95  }

105  std::pair<matrix_t<Number>, vector_t<Number>> reduceLinTrans( const matrix_t<Number>& _a, const vector_t<Number>& _b, std::size_t _power ) {
106  std::size_t powerOfTwo = carl::pow( 2, _power );
107  // first compute the new b
108  vector_t<Number> bTrans = _b;
109  matrix_t<Number> aTrans = _a;
110  for ( std::size_t i = 1; i < powerOfTwo; i++ ) {
111  bTrans = _a * bTrans + _b;
112  }
113  // now compute a^i efficiently
114  for ( std::size_t i = 0; i < _power; i++ ) {
115  aTrans = aTrans * aTrans;
116  }
117  return std::make_pair( aTrans, bTrans );
118  }

385  bool checkTreeValidity() const {
386 #ifndef NDEBUG
387  using Node = std::shared_ptr<SupportFunctionContent<Number, Setting>>;
388  using Res = bool;
389  std::vector<Node> callStack;
390  std::vector<std::pair<int, std::vector<Res>>> resultStack; // The first value is an iterator to the calling frame
391 
392  callStack.push_back( getThis() );
393  resultStack.push_back( std::make_pair( -1, std::vector<Res>() ) );
394 
395  while ( !callStack.empty() ) {
396  Node cur = callStack.back();
397 
398  if ( cur->originCount() == 0 ) {
399  // Do computation and write results in case recursion ends.
400 
401  std::pair<int, std::vector<Res>> currentResult = resultStack.back();
402 
403  // update result
404  // special case: When the node is a leaf, we directly return the result.
405  if ( currentResult.first == -1 ) {
406  // first item is also top.
407  return true;
408  }
409 
410  switch ( cur->type() ) {
411  case SF_TYPE::INFTY_BALL:
412  case SF_TYPE::TWO_BALL:
413  case SF_TYPE::POLY:
414  case SF_TYPE::BOX:
415  case SF_TYPE::ELLIPSOID: {
416  resultStack.at( currentResult.first ).second.push_back( true );
417  break;
418  }
419  default:
420  assert( false );
421  }
422 
423  // leave recursive call.
424  callStack.pop_back();
425  resultStack.pop_back();
426 
427  } else {
428  // Detect, if this call is finished or new.
429  if ( resultStack.back().second.size() == cur->originCount() ) {
430  // the call is finished, perform accumulating operations and forward result.
431 
432  // accumulate results
433  Res accumulatedResult = 0;
434  switch ( cur->type() ) {
435  case SF_TYPE::INTERSECT:
436  case SF_TYPE::SUM:
437  case SF_TYPE::LINTRAFO:
438  case SF_TYPE::UNITE:
439  case SF_TYPE::SCALE:
440  case SF_TYPE::PROJECTION:
441  accumulatedResult = true;
442  for ( auto val : resultStack.back().second ) {
443  if ( !val ) {
444  accumulatedResult = false;
445  break;
446  }
447  }
448  break;
449  case SF_TYPE::NONE: {
450  std::cout << __func__ << ": SF Type not properly initialized!" << std::endl;
451  assert( false );
452  break;
453  }
454  default:
455  std::cout << __func__ << ": SF Type unexpected!" << std::endl;
456  assert( false );
457  }
458 
459  if ( resultStack.back().first == -1 ) {
460  // we reached the top
461  return accumulatedResult;
462  }
463 
464  resultStack.at( resultStack.back().first ).second.push_back( accumulatedResult );
465 
466  // delete result frame and close recursive call
467  callStack.pop_back();
468  resultStack.pop_back();
469 
470  } else {
471  // this is the branch for calling recursively
472 
473  // here we create the new stack levels. As the parameters are all the same, we do not care for their order (could be extended).
474  int callingFrame = int( callStack.size() ) - 1;
475 
476  switch ( cur->type() ) {
477  case SF_TYPE::SUM: {
478  callStack.push_back( cur->summands()->rhs );
479  callStack.push_back( cur->summands()->lhs );
480  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
481  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
482  break;
483  }
484  case SF_TYPE::INTERSECT: {
485  callStack.push_back( cur->intersectionParameters()->rhs );
486  callStack.push_back( cur->intersectionParameters()->lhs );
487  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
488  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
489  break;
490  }
491  case SF_TYPE::LINTRAFO: {
492  callStack.push_back( cur->linearTrafoParameters()->origin );
493  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
494  break;
495  }
496  case SF_TYPE::SCALE: {
497  callStack.push_back( cur->scaleParameters()->origin );
498  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
499  break;
500  }
501  case SF_TYPE::UNITE: {
502  for ( unsigned i = 0; i < cur->unionParameters()->items.size(); ++i ) {
503  callStack.push_back( cur->unionParameters()->items.at( i ) );
504  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
505  }
506  break;
507  }
508  case SF_TYPE::POLY:
509  case SF_TYPE::INFTY_BALL:
510  case SF_TYPE::TWO_BALL:
511  case SF_TYPE::ELLIPSOID:
512  case SF_TYPE::BOX:
513  case SF_TYPE::ZONOTOPE: {
514  assert( false );
515  break;
516  }
517  case SF_TYPE::PROJECTION: {
518  callStack.push_back( cur->projectionParameters()->origin );
519  resultStack.push_back( std::make_pair( callingFrame, std::vector<Res>() ) );
520  break;
521  }
522  case SF_TYPE::NONE:
523  default:
524  assert( false );
525  }
526  }
527  }
528  }
529 
530  assert( false );
531  std::cout << "THIS SHOULD NOT HAPPEN." << std::endl;
532  return false;
533 #endif
534  // in case this function is not put into an assertion, always return true in productive code.
535  return true;
536  }

704  std::pair<bool, std::size_t> hasTrafo( std::shared_ptr<const lintrafoParameters<Number>>& resNode, const matrix_t<Number>& A, const vector_t<Number>& b, std::size_t depth ) const {
705  switch ( mType ) {
706  case SF_TYPE::SUM: {
707  std::pair<bool, std::size_t> res = summands()->lhs->hasTrafo( resNode, A, b, depth + 1 );
708  if ( !res.first ) {
709  res = summands()->rhs->hasTrafo( resNode, A, b, depth + 1 );
710  }
711  return res;
712  }
713  case SF_TYPE::INTERSECT: {
714  std::pair<bool, std::size_t> res = intersectionParameters()->lhs->hasTrafo( resNode, A, b, depth + 1 );
715  if ( !res.first ) {
716  res = intersectionParameters()->rhs->hasTrafo( resNode, A, b, depth + 1 );
717  }
718  return res;
719  }
720  case SF_TYPE::LINTRAFO: {
721  if ( linearTrafoParameters()->parameters->matrix() == A && linearTrafoParameters()->parameters->vector() == b ) {
722  resNode = linearTrafoParameters()->parameters;
723  return std::make_pair( true, depth + 1 );
724  }
725  return linearTrafoParameters()->origin->hasTrafo( resNode, A, b, depth + 1 );
726  }
727  case SF_TYPE::SCALE: {
728  return scaleParameters()->origin->hasTrafo( resNode, A, b, depth + 1 );
729  }
730  case SF_TYPE::UNITE: {
731  for ( const auto& item : unionParameters()->items ) {
732  auto tmp = item->hasTrafo( resNode, A, b, depth + 1 );
733  if ( tmp.first ) {
734  return tmp;
735  }
736  }
737  return std::make_pair( false, 0 );
738  }
739  case SF_TYPE::POLY:
740  case SF_TYPE::INFTY_BALL:
741  case SF_TYPE::TWO_BALL:
742  case SF_TYPE::ELLIPSOID:
743  case SF_TYPE::BOX:
744  case SF_TYPE::ZONOTOPE: {
745  return std::make_pair( false, 0 );
746  }
747  case SF_TYPE::PROJECTION: {
748  return projectionParameters()->origin->hasTrafo( resNode, A, b, depth + 1 );
749  break;
750  }
751  case SF_TYPE::NONE:
752  default:
753  assert( false );
754  return std::make_pair( false, 0 );
755  }

758  std::vector<SF_TYPE> collectLevelEntries( unsigned level ) const {
759  //std::cout << __func__ << ": level: " << level << std::endl;
760  std::vector<SF_TYPE> items;
761  if ( level == 0 ) {
762  items.push_back( this->mType );

764  //std::cout << "items size: " << items.size() << std::endl;

766  }

769  switch ( mType ) {
770  case SF_TYPE::LINTRAFO: {
771  //std::cout << "Current: lintrafor" << std::endl;
772  std::vector<SF_TYPE> tmpItems = linearTrafoParameters()->origin->collectLevelEntries( level - 1 );

781  }

783  //std::cout << "Current: projection" << std::endl;
784  std::vector<SF_TYPE> tmpItems = projectionParameters()->origin->collectLevelEntries( level - 1 );
785  items.insert( items.end(), tmpItems.begin(), tmpItems.end() );
786  return items;
787  }
788  case SF_TYPE::SUM: {
789  //std::cout << "Current: sum" << std::endl;
790  std::vector<SF_TYPE> lhsItems = summands()->lhs->collectLevelEntries( level - 1 );
791  std::vector<SF_TYPE> rhsItems = summands()->rhs->collectLevelEntries( level - 1 );
792  items.insert( items.end(), lhsItems.begin(), lhsItems.end() );
793  items.insert( items.end(), rhsItems.begin(), rhsItems.end() );
794  return items;
795  }
796  case SF_TYPE::UNITE: {
797  //std::cout << "Current: union" << std::endl;
798  std::vector<SF_TYPE> res;
799  for ( const auto& item : unionParameters()->items ) {
800  std::vector<SF_TYPE> tmp = item->collectLevelEntries( level - 1 );
801  res.insert( res.end(), tmp.begin(), tmp.end() );
802  }
803  return res;
804  }
805  case SF_TYPE::INTERSECT: {

808  std::vector<SF_TYPE> rhsItems = intersectionParameters()->rhs->collectLevelEntries( level - 1 );
809  items.insert( items.end(), lhsItems.begin(), lhsItems.end() );
810  items.insert( items.end(), rhsItems.begin(), rhsItems.end() );
811  return items;
812  }
813  default:
814  return std::vector<SF_TYPE>();
