299  template <typename N = Number, carl::DisableIf<carl::is_float<N>> = carl::dummy>
300  void reduceNumberRepresentation( std::size_t limit = fReach_DENOMINATOR ) const {
301  if ( !mVertices.empty() ) {
302  // determine barycenter to set rounding directions
303  std::size_t dimension = std::size_t( mVertices.begin()->rawCoordinates().rows() );
304  vector_t<Number> barycenter = vector_t<Number>::Zero( dimension );
305  for ( const auto& vertex : mVertices ) {
306  barycenter = barycenter + ( vertex.rawCoordinates() / Number( mVertices.size() ) );
307  }
308  for ( auto& vertex : mVertices ) {
309  // distance vector to the barycenter - each component now determines its own rounding direction.
310  vector_t<Number> roundingDirections = vertex.rawCoordinates() - barycenter;
311  for ( std::size_t d = 0; d < dimension; ++d ) {
312  assert( d < vertex.dimension() );
313  // determine, which is larger: numerator or denominator
314  Number denom = carl::getDenom( vertex.at( d ) );
315  Number num = carl::getNum( vertex.at( d ) );
316  if ( denom > limit && carl::abs( num ) > limit ) {
317  if ( denom > num ) {
318  Number newNum = num / denom * limit;
319  if ( roundingDirections( d ) >= 0 ) { // round up
320  newNum = carl::ceil( newNum );
321  assert( newNum / limit >= vertex.at( d ) );
322  } else { // round down
323  newNum = carl::floor( newNum );
324  assert( newNum / limit <= vertex.at( d ) );
325  }
326  vertex[d] = newNum / limit;
327  } else { // num >= denom
328  Number newDenom = denom / num * limit;
329  if ( roundingDirections( d ) >= 0 ) { // round up
330  newDenom = carl::floor( newDenom );
331  assert( limit / newDenom >= vertex.at( d ) );
332  } else { // round down
333  newDenom = carl::ceil( newDenom );
334  assert( limit / newDenom <= vertex.at( d ) );
335  }
336  vertex[d] = limit / newDenom;
337  }
338  }
339  }
340  }
341  }
