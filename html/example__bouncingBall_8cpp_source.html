1 /**
2  * \example example_bouncingBall.cpp
3  * An example file showing how to construct the automaton for a bouncing ball system and afterwards determine the set of reachable states
4  * for this system using H-polytopes as a state set representation.
5  */
6 
7 #include "config.h"
8 #include "datastructures/HybridAutomaton/HybridAutomaton.h"
9 #include "datastructures/Point.h"
10 #include "representations/GeometricObject.h"
11 #include "algorithms/reachability/Reach.h"
12 #include "util/plotting/Plotter.h"
13 
14 int main()
15 {
16  using namespace hypro;
17 
18  // typedefs for simplification.
19  typedef mpq_class Number;
20  typedef hypro::HPolytope<Number> Representation;
21 
22  // create the discrete structure of the automaton and the automaton itself.
23  std::unique_ptr<Location<Number>> loc1 = std::make_unique<Location<Number>>();
24  std::unique_ptr<Transition<Number>> trans = std::make_unique<Transition<Number>>();
25  HybridAutomaton<Number> bBallAutomaton = HybridAutomaton<Number>();
26 
27  // matrix defining the flow (note: 3rd dimension for constant parts).
28  matrix_t<Number> flowMatrix = matrix_t<Number>(3,3);
29 
30  // location parameter setting.
31  // creation of the invariant.
32  matrix_t<Number> invariantMat = matrix_t<Number>(1,2);
33  vector_t<Number> invariantVec = vector_t<Number>(1);
34 
35  invariantVec(0) = Number(0);
36 
37  invariantMat(0,0) = Number(-1);
38  invariantMat(0,1) = Number(0);
39 
40  Condition<Number> inv(invariantMat,invariantVec);
41 
42  loc1->setInvariant(inv);
43 
44  // setup flow matrix (3x3, we add an artificial dimension to cope with constants).
45  flowMatrix(0,0) = Number(0);
46  flowMatrix(0,1) = Number(1);
47  flowMatrix(0,2) = Number(0);
48  flowMatrix(1,0) = Number(0);
49  flowMatrix(1,1) = Number(0);
50  flowMatrix(1,2) = Number(carl::rationalize<Number>(-9.81));
51  flowMatrix(2,0) = Number(0);
52  flowMatrix(2,1) = Number(0);
53  flowMatrix(2,2) = Number(0);
54 
55  loc1->setFlow(flowMatrix);
56 
57  // setup of the transition.
58  // guard
59  Condition<Number> guard;
60  matrix_t<Number> guardMat = matrix_t<Number>(3,2);
61  vector_t<Number> guardVec = vector_t<Number>(3);
62 
63  guardVec(0) = Number(0);
64  guardVec(1) = Number(0);
65  guardVec(2) = Number(0);
66 
67  guardMat(0,0) = Number(1);
68  guardMat(0,1) = Number(0);
69  guardMat(1,0) = Number(-1);
70  guardMat(1,1) = Number(0);
71  guardMat(2,0) = Number(0);
72  guardMat(2,1) = Number(1);
73 
74  guard.setMatrix(guardMat);
75  guard.setVector(guardVec);
76 
77  // reset function
78  Reset<Number> reset;
79  vector_t<Number> constantReset = vector_t<Number>(2,1);
80  matrix_t<Number> linearReset = matrix_t<Number>(2,2);
81 
82  constantReset(0) = Number(0);
83  constantReset(1) = Number(0);
84 
85  linearReset(0,0) = Number(1);
86  linearReset(0,1) = Number(0);
87  linearReset(1,0) = Number(0);
88  linearReset(1,1) = Number(carl::rationalize<Number>(-0.9));
89 
90  reset.setVector(constantReset);
91  reset.setMatrix(linearReset);
92 
93  // setup transition
94  trans->setAggregation(Aggregation::parallelotopeAgg);
95  trans->setGuard(guard);
96  trans->setSource(loc1);
97  trans->setTarget(loc1);
98  trans->setReset(reset);
99 
100 
101  // add defined location and transition to the automaton.
102  loc1->addTransition(std::move(trans));
103 
104  // create Box holding the initial set.
105  matrix_t<Number> boxMat = matrix_t<Number>(4,2);
106  vector_t<Number> boxVec = vector_t<Number>(4);
107 
108  boxVec(0) = Number(carl::rationalize<Number>(10.2));
109  boxVec(1) = Number(-10);
110  boxVec(2) = Number(0);
111  boxVec(3) = Number(0);
112 
113  boxMat(0,0) = Number(1);
114  boxMat(0,1) = Number(0);
115  boxMat(1,0) = Number(-1);
116  boxMat(1,1) = Number(0);
117  boxMat(2,0) = Number(0);
118  boxMat(2,1) = Number(1);
119  boxMat(3,0) = Number(0);
120  boxMat(3,1) = Number(-1);
121 
122  // create initial state.
123  bBallAutomaton.addInitialState(loc1.get(), Condition<Number>(boxMat,boxVec));
124 
125  bBallAutomaton.addLocation(std::move(loc1));
126 
127  // vector of sets to collect flowpipes (which are again a vector of sets).
128  std::vector<std::vector<Representation>> flowpipes;
129 
130  // instanciate reachability analysis algorithm.
131  hypro::reachability::Reach<Number,hypro::reachability::ReachSettings,hypro::State_t<Number>> reacher(bBallAutomaton);
132  hypro::ReachabilitySettings settings = reacher.settings();
133  settings.timeStep = carl::convert<double,Number>(0.01);
134  settings.timeBound = Number(3);
135  settings.jumpDepth = 3;
136  reacher.setSettings(settings);
137  reacher.setRepresentationType(Representation::type());
138 
139  // perform reachability analysis.
140  auto flowpipeIndices = reacher.computeForwardReachability();
141 
142  // plot flowpipes.
143  Plotter<Number>& plotter = Plotter<Number>::getInstance();
144  plotter.setFilename("bouncingBall");
145  for(auto& indexPair : flowpipeIndices){
146  std::vector<hypro::State_t<Number>> flowpipe = indexPair.second;
147  // Plot single flowpipe
148  for(auto& set : flowpipe) {
149  std::vector<Point<Number>> points = set.vertices();
150  if(!points.empty() && points.size() > 2) {
151  for(auto& point : points) {
152  point.reduceDimension(2);
153  }
154  plotter.addObject(points);
155  points.clear();
156  }
157  }
158  }
159 
160  // write output.
161  plotter.plot2d();
162 
163  return 0;
164 }
