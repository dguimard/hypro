113  void addContinuousState(const std::vector<matrix_t<Number>>& _constraint, unsigned id, std::vector<State<Number,ConstraintSet<Number>>>& _states) {
114  TRACE("hypro.parser","Add new continuous state.");
115  if(_constraint.size() > 0) {
116  TRACE("hypro.parser","Add continuous state for location " << id);
117  bool found = false;
118  // due to the creation of initial states, which uses this function as well and the occurrence of duplicates in the initial states, we need the last instance of a matching state.
119  for(auto stateIt = _states.rbegin(); stateIt != _states.rend(); ++stateIt) {
120  if(stateIt->getLocation()->getId() == id){
121  TRACE("hypro.parser","stateIt->already exists.");
122  found = true;
123  unsigned constraintsNum = 0;
124  if(stateIt->getNumberSets() > 0) {
125  constraintsNum = unsigned(boost::get<ConstraintSet<Number>>(stateIt->getSet()).matrix().rows());
126  TRACE("hypro.parser","current constraints: " << boost::get<ConstraintSet<Number>>(stateIt->getSet()).matrix());
127  }
128  unsigned dimension = constraintsNum > 0 ? unsigned(boost::get<ConstraintSet<Number>>(stateIt->getSet()).matrix().cols()) : unsigned((_constraint.begin()->cols())-1);
129  ConstraintSet<Number> set = constraintsNum > 0 ? boost::get<ConstraintSet<Number>>(stateIt->getSet()) : ConstraintSet<Number>();
130  TRACE("hypro.parser","Resize to " << constraintsNum+_constraint.size() << " x " << dimension);
131  set.rMatrix().conservativeResize(constraintsNum+Eigen::Index(_constraint.size()), dimension);
132  set.rVector().conservativeResize(constraintsNum+Eigen::Index(_constraint.size()));
133  TRACE("hypro.parser","State already existing with " << constraintsNum << " rows.");
134  for(const auto& row : _constraint) {
135  assert(row.rows() == 1);
136  assert(row.cols() > 0);
137  TRACE("hypro.parser","add row " << constraintsNum);
138  TRACE("hypro.parser","row properties: " << row.rows() << " , " << row.cols());
139  TRACE("hypro.parser","set properties: " << set.matrix().rows() << " , " << set.matrix().cols());
140  set.rMatrix().row(constraintsNum) = row.block(0,0,1,row.cols()-1);
141  //TRACE("hypro.parser"," " << set.matrix().row(constraintsNum) << " <= " );
142  set.rVector()(constraintsNum) = -row(0,row.cols()-1);
143  //TRACE("hypro.parser",set.vector()(constraintsNum));
144  ++constraintsNum;
145  }
146  stateIt->setSet(set);
147  TRACE("hypro.parser","New constraints: " << boost::get<ConstraintSet<Number>>(stateIt->getSet()).matrix());
148  break;
149  }
150  }
151  if(!found){
152  State<Number,ConstraintSet<Number>> s(mLocationManager.location(id));
153  ConstraintSet<Number> set(matrix_t<Number>(_constraint.size(), _constraint.begin()->cols()-1), vector_t<Number>(_constraint.size()));
154  unsigned constraintsNum = 0;
155  TRACE("hypro.parser","State newly created with " << _constraint.size() << " rows.");
156  for(const auto& row : _constraint) {
157  assert(row.rows() == 1);
158  TRACE("hypro.parser","add row " << constraintsNum);
159  set.rMatrix().row(constraintsNum) = row.block(0,0,1,row.cols()-1);
160  //TRACE("hypro.parser"," " << set.matrix().row(constraintsNum) << " <= ");
161  set.rVector()(constraintsNum) = -row(0,row.cols()-1);
162  //TRACE("hypro.parser",set.vector()(constraintsNum));
163  ++constraintsNum;
164  }
165  s.setSet(set);
166  _states.emplace_back(s);
167  }
168  }

219  void insertModes(const std::vector<std::pair<std::string, Location<Number>*>>& _in) {
220  for(const auto& modePair : _in) {
221  TRACE("hypro.parser","Found mode " << modePair.first << " mapped to " << modePair.second->getId());
222  TRACE("hypro.parser","Mode: " << *modePair.second);
223 
224  if(mModes.find(modePair.first) == nullptr) {
225  mModes.add(modePair.first, modePair.second->getId());
226  mModeIds.push_back(modePair.second->getId());
227  TRACE("hypro.parser","Add new mode " << modePair.first << " mapped to id " << modePair.second->getId());
228  } else {
229  TRACE("hypro.parser","Overwrite mode " << modePair.first << " mapped to id " << mModes.at(modePair.first) << " with mode mapped to id " << modePair.second->getId());
230  unsigned redundantId = mModes.at(modePair.first);
231  mModes.at(modePair.first) = modePair.second->getId();
232  mLocationManager.erase(redundantId);
233  // update mModeIds
234  auto modeIdIt = mModeIds.begin();
235  while(modeIdIt != mModeIds.end() && *modeIdIt != redundantId) { ++modeIdIt; }
236  assert(modeIdIt != mModeIds.end());
237  if(modeIdIt != mModeIds.end()) {
238  mModeIds.erase(modeIdIt);
239  }
240  mModeIds.push_back(modePair.second->getId());
241  }
242  }
