102  //std::cout << this << " Request parameter set for exponent " << exponent << std::endl;
103  if(parameters.find(exponent) != parameters.end()){

106  // create new parameters

149  void createNextReduct() const {
150  // the last created reduction pair is at the back, as a std::map is per default sorted ascending
151  assert(parameters.size() > 0);
152  unsigned powerOfTwo = carl::pow(2, power);
153  // first compute the new b
154  vector_t<Number> bTrans = (--parameters.end())->second.second;
155  matrix_t<Number> aTrans = (--parameters.end())->second.first;
156  //std::cout << "Starting from exponent " << (--matrices.end())->first << std::endl;
157  unsigned exponent = ((--parameters.end())->first) * powerOfTwo;
158  //std::cout << "New exponent: " << exponent << std::endl;
159  // accumulate b
160  for (std::size_t i = 1; i < powerOfTwo ; i++){
161  // Note: aTrans hasn't changed yet -> we can use it for transformation.
162  bTrans = aTrans*bTrans + (--parameters.end())->second.second;
163  }
164  // accumulate A
165  for (std::size_t i = 0; i < power; i++){
166  aTrans = aTrans*aTrans;
167  }
168  parameters[exponent] = std::make_pair(aTrans, bTrans);
169  }
